public static void unescape(ZapposJson zapposJson, CharSequence value, Writer writer) throws IOException {
    boolean inescape = false;
    for (int i = 0; i < value.length(); i++) {
        char c = value.charAt(i);
        if (c != '\\' && !inescape) {
            writer.append(c);
            continue;
        } else if (c == '\\') {
            inescape = true;
            continue;
        }
        switch(c) {
            case 'u':
                writer.append(String.valueOf(Integer.parseInt(value.subSequence(i, 4).toString(), 16)));
                i += 4;
                break;
            case 't':
                writer.append('\t');
                break;
            case 'b':
                writer.append('\b');
                break;
            case 'n':
                writer.append('\n');
                break;
            case 'r':
                writer.append('\r');
                break;
            case 'f':
                writer.append('\f');
                break;
            case '\n':
            case '\'':
            case '"':
            case '\\':
            default:
                writer.append(c);
        }
        inescape = false;
    }
}
