public static String randomAlphabetic(int n) {
    char[] c = new char[n];
    for (int i = 0; i < n; i++) {
        int j = RANDOM.nextInt(26);
        c[i] = RANDOM.nextBoolean() ? BIG_LETTERS[j] : SMALL_LETTERS[j];
    }
    return new String(c);
}
